cmake_minimum_required(VERSION 2.8)

SET(PROJ_NAME "AdonEngineProject")
SET(PROJ_PATH ${CMAKE_SOURCE_DIR})
SET(PROJ_OUT_PATH ${CMAKE_BINARY_DIR})
SET(PROJ_SOURCE ${PROJ_PATH}/Source)
SET(PROJ_GRAPHICS ${PROJ_PATH}/Source/Graphics)
SET(PROJ_LIBRARIES ${PROJ_PATH}/libs/GLEW/glew32.lib ${PROJ_PATH}/libs/GLFW/glfw3.lib)
SET(PROJ_INCLUDES "libs/GLEW" "libs/GLFW" "Source/Graphics/Header" "Source/Graphics")

FIND_PACKAGE(OpenGL)

project(${PROJ_NAME})

INCLUDE_DIRECTORIES(${PROJ_INCLUDES})
LINK_DIRECTORIES(${PROJ_LIBRARIES})

FILE(GLOB SRC ${PROJ_GRAPHICS}/Source/*cc)
FILE(GLOB HEAD ${PROJ_GRAPHICS}/Header/*h)
LIST(APPEND SRCSGRAPHICS ${SRC} ${HEAD})
SOURCE_GROUP(Graphics FILES ${SRCSGRAPHICS})

FILE(GLOB SRC ${PROJ_SOURCE}/*cc)
FILE(GLOB HEAD ${PROJ_SOURCE}/*h)
LIST(APPEND SRCSMAIN ${SRC} ${HEAD})
SOURCE_GROUP(Main FILES ${SRCSMAIN})

LIST(APPEND ALLSRCS ${SRCSMAIN} ${SRCSGRAPHICS})

ADD_EXECUTABLE(${PROJ_NAME} ${ALLSRCS})

TARGET_LINK_LIBRARIES(${PROJ_NAME} ${PROJ_LIBRARIES} ${OPENGL_LIBRARIES})
